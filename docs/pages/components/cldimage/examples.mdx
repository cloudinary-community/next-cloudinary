## Image Overlays

<ImageGrid>
  <li>
    <CldImage
      width="960"
      height="600"
      src="images/turtle"
      sizes="100vw"
      overlays={[{
        publicId: 'images/earth',
        position: {
          x: 50,
          y: 50,
          gravity: 'north_west',
        },
        effects: [
          {
            crop: 'fill',
            gravity: 'auto',
            width: 500,
            height: 500
          }
        ]
      }]}
    />

    ### Overlay Image by Public ID

    ```jsx
    overlays={[{
      publicId: 'images/earth',
      position: {
        x: 10,
        y: 10,
        gravity: 'north_west',
      },
      effects: [
        {
          crop: 'fill',
          gravity: 'auto',
          width: 500,
          height: 500
        }
      ]
    }]}
    ```
  </li>
</ImageGrid>
